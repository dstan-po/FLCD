	digit = "0" | "1" | ... | "9"

	1. Integer:
		integer = "0" | ["+" | "-"]non_zero{digit}

		non_zero = "1" | "2" | ... | "9"

	2. Charchter:
		charachter = "'"letter"'" | "'"digit"'"

		constchar = """string"""
